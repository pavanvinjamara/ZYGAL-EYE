3. MongoDB Connection (native driver, no Mongoose)

// backend/src/config/db.js
const { MongoClient } = require('mongodb');
require('dotenv').config();

const uri = process.env.MONGO_URI || 'mongodb://localhost:27017';
const dbName = process.env.MONGO_DB || 'iifl_eye';

let client;
let db;

async function connectDB() {
  if (db) return db;
  client = new MongoClient(uri, { maxPoolSize: 20 });
  await client.connect();
  db = client.db(dbName);
  console.log(`MongoDB connected: ${dbName}`);
  return db;
}

function getDB() {
  if (!db) throw new Error('DB not initialized. Call connectDB() first.');
  return db;
}

async function closeDB() {
  if (client) await client.close();
}

module.exports = { connectDB, getDB, closeDB };

5. Creating collections + indexes (script, run once)

// backend/scripts/createIndexes.js
const { connectDB, closeDB } = require('../src/config/db');
const COL = require('../src/db/collections');
const { vendorSchema } = require('../src/models/vendor.model');
const { userSchema } = require('../src/models/user.model');
const { ticketSchema } = require('../src/models/ticket.model');
const { penaltySchema } = require('../src/models/penalty.model');
const { sopViolationSchema, otpLogSchema } = require('../src/models/sopViolation.model');
const { misUploadSchema } = require('../src/models/misUpload.model');
const { bleAssetSchema } = require('../src/models/bleAsset.model');
const { footageRequestSchema } = require('../src/models/footageRequest.model');

async function run() {
  const db = await connectDB();

  const collectionsToCreate = [
    [COL.VENDORS, vendorSchema],
    [COL.USERS, userSchema],
    [COL.TICKETS, ticketSchema],
    [COL.PENALTIES, penaltySchema],
    [COL.SOP_VIOLATIONS, sopViolationSchema],
    [COL.OTP_LOGS, otpLogSchema],
    [COL.MIS_UPLOADS, misUploadSchema],
    [COL.BLE_ASSETS, bleAssetSchema],
    [COL.FOOTAGE_REQUESTS, footageRequestSchema],
  ];

  const existing = (await db.listCollections().toArray()).map(c => c.name);

  for (const [name, schema] of collectionsToCreate) {
    if (!existing.includes(name)) {
      await db.createCollection(name, { validator: schema });
      console.log(`Created collection: ${name}`);
    }
  }

  // Indexes
  await db.collection(COL.USERS).createIndex({ email: 1 }, { unique: true });
  await db.collection(COL.VENDORS).createIndex({ shortCode: 1 }, { unique: true });
  await db.collection(COL.TICKETS).createIndex({ ticketRef: 1 }, { unique: true });
  await db.collection(COL.TICKETS).createIndex({ vendorId: 1, status: 1 });
  await db.collection(COL.TICKETS).createIndex({ vendorId: 1, slaStatus: 1 });
  await db.collection(COL.PENALTIES).createIndex({ vendorId: 1, month: 1 });
  await db.collection(COL.SOP_VIOLATIONS).createIndex({ vendorId: 1, status: 1 });
  await db.collection(COL.OTP_LOGS).createIndex({ vendorId: 1, issuedAt: -1 });
  await db.collection(COL.MIS_UPLOADS).createIndex({ vendorId: 1, type: 1, uploadedAt: -1 });
  await db.collection(COL.BLE_ASSETS).createIndex({ assetTag: 1 }, { unique: true });
  await db.collection(COL.BLE_ASSETS).createIndex({ vendorId: 1, status: 1 });
  await db.collection(COL.FOOTAGE_REQUESTS).createIndex({ requestRef: 1 }, { unique: true });

  console.log('Indexes created.');
  await closeDB();
}

run().catch(err => { console.error(err); process.exit(1); });

6. Repository pattern example (no ORM, raw driver)

// backend/src/repositories/ticket.repo.js
const { getDB } = require('../config/db');
const { ObjectId } = require('mongodb');
const COL = require('../db/collections');

async function findByVendor(vendorId, filter = {}) {
  const db = getDB();
  return db.collection(COL.TICKETS)
    .find({ vendorId: new ObjectId(vendorId), ...filter })
    .sort({ openedAt: -1 })
    .toArray();
}

async function findByRef(ticketRef) {
  const db = getDB();
  return db.collection(COL.TICKETS).findOne({ ticketRef });
}

async function create(ticket) {
  const db = getDB();
  const doc = { ...ticket, createdAt: new Date(), updatedAt: new Date() };
  const result = await db.collection(COL.TICKETS).insertOne(doc);
  return { _id: result.insertedId, ...doc };
}

async function addTimelineEntry(ticketRef, entry) {
  const db = getDB();
  return db.collection(COL.TICKETS).updateOne(
    { ticketRef },
    { $push: { timeline: entry }, $set: { updatedAt: new Date() } }
  );
}

async function resolve(ticketRef, closureReport) {
  const db = getDB();
  return db.collection(COL.TICKETS).updateOne(
    { ticketRef },
    {
      $set: { status: 'resolved', resolvedAt: new Date(), closureReport, updatedAt: new Date() },
      $push: { timeline: { type: 'done', text: 'Ticket resolved', at: new Date() } },
    }
  );
}

module.exports = { findByVendor, findByRef, create, addTimelineEntry, resolve };

// backend/src/services/auth.service.js
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const userRepo = require('../repositories/user.repo');
const vendorRepo = require('../repositories/vendor.repo');

const JWT_SECRET = process.env.JWT_SECRET;
const JWT_EXPIRY = '12h';

async function login({ email, password, vendorShortCode }) {
  const user = await userRepo.findByEmail(email);
  if (!user || user.status !== 'active') {
    throw Object.assign(new Error('Invalid credentials'), { statusCode: 401 });
  }

  const match = await bcrypt.compare(password, user.passwordHash);
  if (!match) throw Object.assign(new Error('Invalid credentials'), { statusCode: 401 });

  // Vendor users must select the matching company on login
  if (user.vendorId && vendorShortCode) {
    const vendor = await vendorRepo.findByShortCode(vendorShortCode);
    if (!vendor || String(vendor._id) !== String(user.vendorId)) {
      throw Object.assign(new Error('Vendor mismatch'), { statusCode: 401 });
    }
  }

  await userRepo.touchLogin(user._id);

  const token = jwt.sign(
    { sub: String(user._id), role: user.role, vendorId: user.vendorId ? String(user.vendorId) : null },
    JWT_SECRET,
    { expiresIn: JWT_EXPIRY }
  );

  const { passwordHash, ...safeUser } = user;
  return { token, user: safeUser };
}

async function register({ name, email, password, role, vendorId }) {
  const existing = await userRepo.findByEmail(email);
  if (existing) throw Object.assign(new Error('Email already in use'), { statusCode: 409 });

  const passwordHash = await bcrypt.hash(password, 10);
  return userRepo.create({ name, email, passwordHash, role, vendorId: vendorId || null, status: 'invited' });
}

async function changePassword(userId, oldPassword, newPassword) {
  const user = await userRepo.findById(userId);
  if (!user) throw Object.assign(new Error('User not found'), { statusCode: 404 });
  const ok = await bcrypt.compare(oldPassword, user.passwordHash);
  if (!ok) throw Object.assign(new Error('Current password incorrect'), { statusCode: 400 });
  const newHash = await bcrypt.hash(newPassword, 10);
  await userRepo.setPasswordHash(userId, newHash);
}

module.exports = { login, register, changePassword };

// backend/src/middlewares/auth.middleware.js
const jwt = require('jsonwebtoken');

function authenticate(req, res, next) {
  const header = req.headers.authorization || '';
  const token = header.startsWith('Bearer ') ? header.slice(7) : null;
  if (!token) return res.status(401).json({ success: false, message: 'Missing token' });

  try {
    const payload = jwt.verify(token, process.env.JWT_SECRET);
    req.user = payload; // { sub, role, vendorId }
    next();
  } catch (err) {
    return res.status(401).json({ success: false, message: 'Invalid or expired token' });
  }
}

module.exports = { authenticate };

function requireRole(...roles) {
  return (req, res, next) => {
    if (!req.user || !roles.includes(req.user.role)) {
      return res.status(403).json({ success: false, message: 'Forbidden' });
    }
    next();
  };
}

// Ensures vendor users only touch their own vendor's data
function scopeToOwnVendor(req, res, next) {
  if (req.user.vendorId && req.params.vendorId && req.params.vendorId !== req.user.vendorId) {
    return res.status(403).json({ success: false, message: 'Cross-vendor access denied' });
  }
  next();
}

// backend/src/utils/slaCalculator.js
const TARGETS = { P1: 4, P2: 8, P3: 24 }; // hours
const PENALTY_RATE = { P1: 3000, P2: 1500, P3: 500 }; // ₹/hr

function slaTargetHours(priority) { return TARGETS[priority] || 24; }

function computeSlaStatus(openedAt, priority, resolvedAt = null) {
  const target = TARGETS[priority] || 24;
  const elapsedHrs = ((resolvedAt || new Date()) - new Date(openedAt)) / 3.6e6;
  if (elapsedHrs > target) return 'breach';
  if (elapsedHrs > target * 0.75) return 'warn';
  return 'ok';
}

function computePenalty(priority, overageHours) {
  return Math.ceil(overageHours) * (PENALTY_RATE[priority] || 500);
}

module.exports = { slaTargetHours, computeSlaStatus, computePenalty, TARGETS, PENALTY_RATE };

// backend/src/jobs/slaBreachCheck.job.js
const { getDB } = require('../config/db');
const COL = require('../db/collections');
const { computeSlaStatus, computePenalty, slaTargetHours } = require('../utils/slaCalculator');

async function runSlaBreachCheck() {
  const db = getDB();
  const openTickets = await db.collection(COL.TICKETS)
    .find({ status: { $ne: 'resolved' } }).toArray();

  for (const t of openTickets) {
    const newStatus = computeSlaStatus(t.openedAt, t.priority);
    if (newStatus !== t.slaStatus) {
      await db.collection(COL.TICKETS).updateOne(
        { _id: t._id }, { $set: { slaStatus: newStatus, updatedAt: new Date() } }
      );
    }
    if (newStatus === 'breach') {
      const target = slaTargetHours(t.priority);
      const overage = (Date.now() - new Date(t.openedAt)) / 3.6e6 - target;
      const amount = computePenalty(t.priority, overage);
      const month = new Date().toISOString().slice(0, 7);

      const existing = await db.collection(COL.PENALTIES).findOne({ ticketRef: t.ticketRef, month });
      if (!existing) {
        await db.collection(COL.PENALTIES).insertOne({
          vendorId: t.vendorId, ticketRef: t.ticketRef,
          description: `${t.priority} overage on ${t.branchCode}`,
          ruleApplied: `${t.priority} overage penalty`,
          amount, incidentDate: new Date(), status: 'charged', month,
        });
      }
    }
  }
  console.log(`[slaBreachCheck] scanned ${openTickets.length} open tickets`);
}

module.exports = { runSlaBreachCheck };

  }
}

import { defineStore } from 'pinia';
import * as authApi from '@/api/auth.api';

export const useAuthStore = defineStore('auth', {
  state: () => ({
    token: localStorage.getItem('token') || null,
    user: JSON.parse(localStorage.getItem('user') || 'null'),
  }),
  getters: {
    isAuthenticated: (s) => !!s.token,
    isVendorUser: (s) => !!s.user?.vendorId,
  },
  actions: {
    async login(email, password, vendorShortCode) {
      const { data } = await authApi.login({ email, password, vendorShortCode });
      this.token = data.token;
      this.user = data.user;
      localStorage.setItem('token', data.token);
      localStorage.setItem('user', JSON.stringify(data.user));
    },
    logout() {
      this.token = null;
      this.user = null;
      localStorage.removeItem('token');
      localStorage.removeItem('user');
    },
  },
});

What's actually happening

Both logins hit the same endpoint (POST /auth/login) — there's no separate "admin login" URL. What decides what a user can do is two fields on their user document:

Field	Vendor user (e.g. r.kumar@zygal.io)	Internal IIFL user (e.g. soc@iifl.com)
vendorId	set to their vendor's _id	null
role	admin (means "admin of their vendor account")	iifl_soc

The bug: every /api/admin/* route was written as requireRole('iifl_soc', 'admin') — and requireRole only checks the role string, not vendorId. So a vendor's admin user (like r.kumar@zygal.io) — who should only ever see their own vendor's data — currently also passes the check and can hit /admin/vendors, /admin/users/invite, etc. That's the actual source of your confusion: the system currently can't tell the two "admin"s apart.

The fix

Admin-only routes should require requireInternal (i.e. vendorId === null) and role === 'iifl_soc' — never the vendor-scoped admin role.
